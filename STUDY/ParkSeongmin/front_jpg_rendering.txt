  const [imgSrc, setImgSrc] = useState("");
  const ws = new WebSocket("ws://localhost:8081");

  window.addEventListener("keydown", (event) => {
    ws.send(event.keyCode);
  });

  ws.onopen = function () {
    console.log("on open");
  };

  ws.onmessage = function ({ data }) {
    const url = `data:image/jpeg;base64,${data}`;
    setImgSrc(url);
  };
